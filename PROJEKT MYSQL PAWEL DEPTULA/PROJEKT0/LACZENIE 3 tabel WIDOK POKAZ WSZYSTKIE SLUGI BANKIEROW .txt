CREATE VIEW `laczenie 3 pokaz wszyskich pracownikow i slugi pod bankierami` AS SELECT panowie.IMIE,panowie.ROLA,panowie.id_pana,sludzy.IMIE,sludzy.id_szefa,pracownicy.IMIE,pracownicy.SPECJALIZACJA,id_pracownika FROM panowie LEFT JOIN sludzy ON panowie.id_pana=sludzy.id_pana INNER JOIN pracownicy ON  pracownicy.id_szefa=sludzy.id_szefa WHERE ROLA='BANKIER' GROUP BY panowie.IMIE;
